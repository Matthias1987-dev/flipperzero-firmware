REM Target: Windows
REM Dependencies: PowerShell

REM TEST I.O.

STRING powershell

ENTER

ENTER

DELAY 1000
ENTER

DELAY 1000
ENTER

STRING yes
DELAY 1000
ENTER
WAIT_FOR_BUTTON_PRESS
DELAY 1000
STRING 2490
DELAY 1000
